# 타이머(setTimeout)

```javascript
// 전역변수
var initMinute;
var remainSecond;

$(function() {
	clearTime(1); // 세션 만료 시간
	setTimer(); // 타이머 시작
});

function clearTime(min) {
	initMinute = min;
	remainSecond = min * 60;
}

function setTimer() { // 1초 간격으로 타이머 호출
	// hh : mm 으로 남은시간 표기하기 위한 변수
	var remainMinute_ = parseInt(remainSecond/60);
	var remainSecond_ = parseInt(remainSecond%60);

	console.log(remainMinute_);
	console.log(remainSecond_);

	if(remainSecond > 0) {
		$("#timer").empty();
		$("#timer").val(Lpad(remainMinute_, 2) + ":" + +Lpad(remainSecond_, 2)); // hh:mm 표기
		remainSecond--;
		setTimeout("setTimer()", 1000); // 1초간격으로 재귀 호출
	} else {
		alert('글 작성시간이 만료되었습니다. 로그인 페이지로 이동합니다.');
		/* 세션 날리는 컨트롤러로 이동 후 로그인 페이지로 redirect */
	}
}

function Lpad(str, len) {
	str = str + "";
	while(str.length < len) {
		str = "0"+str;
	}
	return str;
}
```
